order:
1. clear current global_cats_spellfix

DELETE FROM global_cats_spellfix;

2. re-add global_cats_spellfix ranks from existing data

WITH RECURSIVE split(id, global_cats, str) AS (
    SELECT id, '', global_cats||',' 
    FROM Links
    UNION ALL SELECT
    id,
    substr(str, 0, instr(str, ',')),
    substr(str, instr(str, ',') + 1)
    FROM split
    WHERE str != ''
)
INSERT INTO global_cats_spellfix (word, rank)
SELECT global_cats, count(global_cats) as count
FROM split
WHERE global_cats != ""
  AND id IN (
    SELECT id 
    FROM split 
  )
GROUP BY global_cats
ORDER BY count DESC, LOWER(global_cats) ASC;
